Fonction plusPetitPlusGrand(
    in nb1:N,
    in nb2:N
):N
    variables locales:
        var plusPetit:B <-0
    fin variables locales
    Si nb1<nb2
        alors
            plusPetit<-True
    Sinon
        plusPetit<-False
    
    retourne plusPetit
Fin plusPetitPlusGrand
